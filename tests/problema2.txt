fun void main() {
    any: entradas = [15, 32, 77, 100, 49, 3, -1];  // simula entrada do usuário

    int: i = 0;
    int: faixa1 = 0;  // [0, 25]
    int: faixa2 = 0;  // [26, 50]
    int: faixa3 = 0;  // [51, 75]
    int: faixa4 = 0;  // [76, 100]
    int: atual;
    if(true){
        print("Isso deve executar");
    }
    if(false){
        print("Isso não deve executar");
    }

    while(i < len(entradas)) {
        atual = entradas[i];

        if(atual < 0) {
            break;
        }

        if(atual >= 0 && atual <= 25) {
            faixa1++;
        } else if(atual >= 26 && atual <= 50) {
            faixa2++;
        } else if(atual >= 51 && atual <= 75) {
            faixa3++;
        } else if(atual >= 76 && atual <= 100) {
            faixa4++;
        }

        i++;
    }

    print("Valores no intervalo [0, 25]:", faixa1);
    print("Valores no intervalo [26, 50]:", faixa2);
    print("Valores no intervalo [51, 75]:", faixa3);
    print("Valores no intervalo [76, 100]:", faixa4);
}
